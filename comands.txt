1 - getting the spring as service

mvn clean package
cp target/*.jar /opt/app.jar
java -jar /opt/app.jar    // test it:

sudo nano /etc/systemd/system/springboot.service 

put them on it  

[Unit]
Description=Spring Boot App
After=network.target

[Service]
User=root
ExecStart=/usr/bin/java -jar /opt/app.jar
SuccessExitStatus=143
Restart=on-failure
RestartSec=5

[Install]
WantedBy=multi-user.targets

close it 

sudo systemctl daemon-reexec   // run it
sudo systemctl daemon-reload
sudo systemctl enable springboot
sudo systemctl start springboot

sudo systemctl status springboot                       // check if it working

------- replace spring service with new one ---------
mvn clean package
sudo cp target/*.jar /opt/app.jar
sudo systemctl restart springboot
sudo systemctl status springboot
